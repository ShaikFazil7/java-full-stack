const box=document.getElementById("chocolate");
let x=100;
let y=100;
const step=10;
document.addEventListener("keydown",function (keys)
{
    if (keys.code=="ArrowDown")
    {
        y+=step;
    }
    if(keys.code=="ArrowUp")
    {
        y-=step;
    }
    box.style.left=x+"px";
    box.style.top=y+"px";
});
const candy= document.getElementById("chocolate");
candy.addEventListener("mouseover",function ()
{
    candy.style.backgroundColor="orange";
    candy.innerText="One Punch Man";
});
candy.addEventListener("mouseout",function() 
{
candy.style.backgroundColor="pink";
candy.innerText="Naruto";
});
